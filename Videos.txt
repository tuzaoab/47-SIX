67 KID

@echo off
powershell -NoProfile -NonInteractive -ExecutionPolicy Bypass -Command "try { $imageUrl = 'https://raw.githubusercontent.com/tuzaoab/67-MANGAS/main/67KID.mp4'; $videoPath = \"$env:TEMP\67KID.mp4\"; Invoke-WebRequest -Uri $imageUrl -OutFile $videoPath -ErrorAction Stop; Invoke-Item $videoPath; Start-Sleep -Seconds 7.5; Remove-Item $videoPath -ErrorAction SilentlyContinue } catch { Write-Error 'Erro ao baixar ou abrir o vídeo: ' + $_.Exception.Message; exit 1 }"
if %ERRORLEVEL% neq 0 (
    echo Erro: Falha ao executar o comando PowerShell.
    pause
)
